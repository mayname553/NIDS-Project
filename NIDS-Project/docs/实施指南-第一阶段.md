创建文件 `backend/data_balancer.py`:
```python
"""
数据平衡模块 - 使用SMOTE解决类别不平衡问题
"""
from imblearn.over_sampling import SMOTE
from collections import Counter
import numpy as np
import logging

logging.basicConfig(level=logging.INFO)
logger = logging.getLogger(__name__)

class DataBalancer:
    def __init__(self, sampling_strategy='auto', k_neighbors=5, random_state=42):
        """
        初始化数据平衡器

        Args:
            sampling_strategy: 采样策略，'auto'表示自动平衡
            k_neighbors: SMOTE算法的K近邻数量
            random_state: 随机种子
        """
        self.smote = SMOTE(
            sampling_strategy=sampling_strategy,
            k_neighbors=k_neighbors,
            random_state=random_state
        )

    def balance(self, X, y):
        """
        平衡数据集

        Args:
            X: 特征矩阵
            y: 标签向量

        Returns:
            X_balanced, y_balanced: 平衡后的数据
        """
        logger.info("="*50)
        logger.info("开始数据平衡处理")
        logger.info("="*50)

        # 统计原始数据分布
        original_dist = Counter(y)
        logger.info(f"原始数据分布: {dict(original_dist)}")
        logger.info(f"原始样本总数: {len(y)}")
        logger.info(f"不平衡比例: {max(original_dist.values()) / min(original_dist.values()):.2f}:1")

        # 应用SMOTE
        X_balanced, y_balanced = self.smote.fit_resample(X, y)

        # 统计平衡后的数据分布
        balanced_dist = Counter(y_balanced)
        logger.info(f"\n平衡后数据分布: {dict(balanced_dist)}")
        logger.info(f"平衡后样本总数: {len(y_balanced)}")
        logger.info(f"增加样本数: {len(y_balanced) - len(y)}")

        logger.info("="*50)
        logger.info("✅ 数据平衡完成")
        logger.info("="*50)

        return X_balanced, y_balanced

if __name__ == '__main__':
```

#### 创建CNN-LSTM模型

创建文件 `backend/models/deep_learning/cnn_lstm_model.py`:
```python
"""
CNN-LSTM混合模型用于网络入侵检测
结合CNN的局部特征提取能力和LSTM的时序建模能力
"""
import torch
import torch.nn as nn
import torch.nn.functional as F

class CNNLSTM(nn.Module):
    def __init__(self, input_dim, hidden_dim=128, num_layers=2, dropout=0.3):
        """
        初始化CNN-LSTM模型

        Args:
            input_dim: 输入特征维度（NSL-KDD为38维）
            hidden_dim: LSTM隐藏层维度
            num_layers: LSTM层数
            dropout: Dropout比例
        """
        super(CNNLSTM, self).__init__()

        self.input_dim = input_dim
        self.hidden_dim = hidden_dim
        self.num_layers = num_layers

        # CNN层 - 提取局部特征
        self.conv1 = nn.Conv1d(in_channels=1, out_channels=64, kernel_size=3, padding=1)
        self.bn1 = nn.BatchNorm1d(64)
        self.conv2 = nn.Conv1d(in_channels=64, out_channels=128, kernel_size=3, padding=1)
        self.bn2 = nn.BatchNorm1d(128)
        self.pool = nn.MaxPool1d(kernel_size=2)

        # 计算CNN输出维度
        cnn_output_dim = input_dim // 2  # 经过一次池化

        # LSTM层 - 捕获时序依赖
        self.lstm = nn.LSTM(
            input_size=128,
            hidden_size=hidden_dim,
            num_layers=num_layers,
            batch_first=True,
            dropout=dropout if num_layers > 1 else 0,
            bidirectional=True  # 双向LSTM
        )

        # 全连接层
        self.fc1 = nn.Linear(hidden_dim * 2 * cnn_output_dim, 256)  # *2因为是双向LSTM
        self.dropout1 = nn.Dropout(dropout)
        self.fc2 = nn.Linear(256, 128)
        self.dropout2 = nn.Dropout(dropout)
        self.fc3 = nn.Linear(128, 2)  # 二分类：正常/攻击

    def forward(self, x):
        """
        前向传播

        Args:
            x: 输入张量 [batch_size, input_dim]

        Returns:
            output: 输出张量 [batch_size, 2]
        """
        batch_size = x.size(0)

        # 调整维度用于CNN: [batch_size, 1, input_dim]
        x = x.unsqueeze(1)

        # CNN特征提取
        x = self.conv1(x)
        x = self.bn1(x)
        x = F.relu(x)

        x = self.conv2(x)
        x = self.bn2(x)
        x = F.relu(x)
        x = self.pool(x)  # [batch_size, 128, input_dim//2]

        # 调整维度用于LSTM: [batch_size, seq_len, features]
        x = x.permute(0, 2, 1)  # [batch_size, input_dim//2, 128]

        # LSTM时序建模
        lstm_out, (h_n, c_n) = self.lstm(x)

        # 展平LSTM输出
        x = lstm_out.contiguous().view(batch_size, -1)

        # 全连接层
        x = self.fc1(x)
        x = F.relu(x)
        x = self.dropout1(x)

        x = self.fc2(x)
        x = F.relu(x)
        x = self.dropout2(x)

        x = self.fc3(x)

        return x

class CNNLSTMTrainer:
    def __init__(self, model, device='cpu', learning_rate=0.001):
        """
        初始化训练器

        Args:
            model: CNN-LSTM模型
            device: 训练设备（'cpu'或'cuda'）
            learning_rate: 学习率
        """
        self.model = model.to(device)
        self.device = device
        self.criterion = nn.CrossEntropyLoss()
        self.optimizer = torch.optim.Adam(model.parameters(), lr=learning_rate)
        self.scheduler = torch.optim.lr_scheduler.ReduceLROnPlateau(
            self.optimizer, mode='min', factor=0.5, patience=5, verbose=True
        )

    def train_epoch(self, train_loader):
        """训练一个epoch"""
        self.model.train()
        total_loss = 0
        correct = 0
        total = 0

        for batch_idx, (data, target) in enumerate(train_loader):
            data, target = data.to(self.device), target.to(self.device)

            # 前向传播
            self.optimizer.zero_grad()
            output = self.model(data)
            loss = self.criterion(output, target)

            # 反向传播
            loss.backward()
            self.optimizer.step()

            # 统计
            total_loss += loss.item()
            _, predicted = output.max(1)
            total += target.size(0)
            correct += predicted.eq(target).sum().item()

        avg_loss = total_loss / len(train_loader)
        accuracy = 100. * correct / total

        return avg_loss, accuracy

    def evaluate(self, test_loader):
        """评估模型"""
        self.model.eval()
        total_loss = 0
        correct = 0
        total = 0

        all_preds = []
        all_targets = []

        with torch.no_grad():
            for data, target in test_loader:
                data, target = data.to(self.device), target.to(self.device)
                output = self.model(data)
                loss = self.criterion(output, target)

                total_loss += loss.item()
                _, predicted = output.max(1)
                total += target.size(0)
                correct += predicted.eq(target).sum().item()

                all_preds.extend(predicted.cpu().numpy())
                all_targets.extend(target.cpu().numpy())

        avg_loss = total_loss / len(test_loader)
        accuracy = 100. * correct / total

        return avg_loss, accuracy, all_preds, all_targets

if __name__ == '__main__':
    # 测试模型
    input_dim = 38  # NSL-KDD特征数
    model = CNNLSTM(input_dim=input_dim)

    # 打印模型结构
    print(model)

    # 测试前向传播
    batch_size = 32
    x = torch.randn(batch_size, input_dim)
    output = model(x)
    print(f"\n输入形状: {x.shape}")
    print(f"输出形状: {output.shape}")
    print("\n✅ CNN-LSTM模型测试成功！")
```

创建文件 `backend/train_cnn_lstm.py`:
```python
"""
训练CNN-LSTM深度学习模型
"""
import torch
import torch.nn as nn
from torch.utils.data import DataLoader, TensorDataset
from sklearn.metrics import accuracy_score, precision_score, recall_score, f1_score, confusion_matrix
import numpy as np
import logging
from tqdm import tqdm
import json
import os

from data_preprocessor import DataPreprocessor
from data_balancer import DataBalancer
from models.deep_learning.cnn_lstm_model import CNNLSTM, CNNLSTMTrainer

logging.basicConfig(level=logging.INFO)
logger = logging.getLogger(__name__)

def prepare_dataloaders(X_train, y_train, X_test, y_test, batch_size=128):
    """准备数据加载器"""
    # 转换为PyTorch张量
    X_train_tensor = torch.FloatTensor(X_train)
    y_train_tensor = torch.LongTensor(y_train)
    X_test_tensor = torch.FloatTensor(X_test)
    y_test_tensor = torch.LongTensor(y_test)

    # 创建数据集
    train_dataset = TensorDataset(X_train_tensor, y_train_tensor)
    test_dataset = TensorDataset(X_test_tensor, y_test_tensor)

    # 创建数据加载器
    train_loader = DataLoader(train_dataset, batch_size=batch_size, shuffle=True)
    test_loader = DataLoader(test_dataset, batch_size=batch_size, shuffle=False)

    return train_loader, test_loader

def train_model(epochs=50, batch_size=128, learning_rate=0.001):
    """训练CNN-LSTM模型"""
    logger.info("="*70)
    logger.info("开始训练CNN-LSTM深度学习模型")
    logger.info("="*70)

    # 1. 数据准备
    logger.info("\n[1/5] 加载和预处理数据...")
    preprocessor = DataPreprocessor()
    X_train, y_train, X_test, y_test = preprocessor.prepare_data(
        '../dataset/KDDTrain+.txt',
        '../dataset/KDDTest+.txt',
        balance=False  # 先不平衡，后面单独处理
    )

    # 2. 数据平衡
    logger.info("\n[2/5] 应用SMOTE数据平衡...")
    balancer = DataBalancer()
    X_train, y_train = balancer.balance(X_train, y_train)

    # 3. 准备数据加载器
    logger.info("\n[3/5] 准备数据加载器...")
    train_loader, test_loader = prepare_dataloaders(
        X_train, y_train, X_test, y_test, batch_size=batch_size
    )
    logger.info(f"训练批次数: {len(train_loader)}")
    logger.info(f"测试批次数: {len(test_loader)}")

    # 4. 创建模型
    logger.info("\n[4/5] 创建CNN-LSTM模型...")
    device = torch.device('cuda' if torch.cuda.is_available() else 'cpu')
    logger.info(f"使用设备: {device}")

    input_dim = X_train.shape[1]
    model = CNNLSTM(input_dim=input_dim, hidden_dim=128, num_layers=2, dropout=0.3)
    trainer = CNNLSTMTrainer(model, device=device, learning_rate=learning_rate)

    # 打印模型参数量
    total_params = sum(p.numel() for p in model.parameters())
    trainable_params = sum(p.numel() for p in model.parameters() if p.requires_grad)
    logger.info(f"模型总参数量: {total_params:,}")
    logger.info(f"可训练参数量: {trainable_params:,}")

    # 5. 训练模型
    logger.info("\n[5/5] 开始训练...")
    logger.info("="*70)

    best_accuracy = 0
    best_epoch = 0
    train_history = {'loss': [], 'accuracy': []}
    val_history = {'loss': [], 'accuracy': []}

    for epoch in range(epochs):
        # 训练
        train_loss, train_acc = trainer.train_epoch(train_loader)
        train_history['loss'].append(train_loss)
        train_history['accuracy'].append(train_acc)

        # 验证
        val_loss, val_acc, _, _ = trainer.evaluate(test_loader)
        val_history['loss'].append(val_loss)
        val_history['accuracy'].append(val_acc)

        # 学习率调整
        trainer.scheduler.step(val_loss)

        # 打印进度
        logger.info(f"Epoch [{epoch+1}/{epochs}] "
                   f"Train Loss: {train_loss:.4f} | Train Acc: {train_acc:.2f}% | "
                   f"Val Loss: {val_loss:.4f} | Val Acc: {val_acc:.2f}%")

        # 保存最佳模型
        if val_acc > best_accuracy:
            best_accuracy = val_acc
            best_epoch = epoch + 1
            torch.save(model.state_dict(), 'model/cnn_lstm_best.pth')
            logger.info(f"✅ 保存最佳模型 (Epoch {best_epoch}, Accuracy: {best_accuracy:.2f}%)")

    logger.info("="*70)
    logger.info(f"训练完成！最佳准确率: {best_accuracy:.2f}% (Epoch {best_epoch})")
    logger.info("="*70)

    # 6. 最终评估
    logger.info("\n开始最终评估...")
    model.load_state_dict(torch.load('model/cnn_lstm_best.pth'))
    _, _, y_pred, y_true = trainer.evaluate(test_loader)

    # 计算详细指标
    accuracy = accuracy_score(y_true, y_pred)
    precision = precision_score(y_true, y_pred, zero_division=0)
    recall = recall_score(y_true, y_pred, zero_division=0)
    f1 = f1_score(y_true, y_pred, zero_division=0)
    cm = confusion_matrix(y_true, y_pred)

    logger.info("\n最终性能指标:")
    logger.info(f"准确率 (Accuracy):  {accuracy:.4f} ({accuracy*100:.2f}%)")
    logger.info(f"精确率 (Precision): {precision:.4f} ({precision*100:.2f}%)")
    logger.info(f"召回率 (Recall):    {recall:.4f} ({recall*100:.2f}%)")
    logger.info(f"F1分数 (F1-Score):  {f1:.4f} ({f1*100:.2f}%)")

    logger.info("\n混淆矩阵:")
    logger.info(f"              预测正常  预测攻击")
    logger.info(f"实际正常      {cm[0][0]:6d}    {cm[0][1]:6d}")
    logger.info(f"实际攻击      {cm[1][0]:6d}    {cm[1][1]:6d}")

    # 保存指标
    metrics = {
        'accuracy': float(accuracy),
        'precision': float(precision),
        'recall': float(recall),
        'f1_score': float(f1),
        'confusion_matrix': cm.tolist(),
        'best_epoch': best_epoch,
        'total_epochs': epochs
    }

    with open('model/cnn_lstm_metrics.json', 'w', encoding='utf-8') as f:
        json.dump(metrics, f, indent=2, ensure_ascii=False)

    logger.info("\n✅ 模型和指标已保存到 model/ 目录")

    return model, metrics

if __name__ == '__main__':
    # 训练模型
    model, metrics = train_model(epochs=50, batch_size=128, learning_rate=0.001)
```

```python
"""
集成学习模型 - 结合Random Forest和CNN-LSTM
"""
import torch
import joblib
import numpy as np
from sklearn.ensemble import VotingClassifier
import logging

logging.basicConfig(level=logging.INFO)
logger = logging.getLogger(__name__)

class EnsembleDetector:
    def __init__(self, rf_model_path, cnn_lstm_model_path, cnn_lstm_config):
        """
        初始化集成检测器

        Args:
            rf_model_path: Random Forest模型路径
            cnn_lstm_model_path: CNN-LSTM模型路径
            cnn_lstm_config: CNN-LSTM模型配置
        """
        # 加载Random Forest
        self.rf_model = joblib.load(rf_model_path)
        logger.info(f"✅ 加载Random Forest模型: {rf_model_path}")

        # 加载CNN-LSTM
        from models.deep_learning.cnn_lstm_model import CNNLSTM
        self.cnn_lstm = CNNLSTM(**cnn_lstm_config)
        self.cnn_lstm.load_state_dict(torch.load(cnn_lstm_model_path))
        self.cnn_lstm.eval()
        logger.info(f"✅ 加载CNN-LSTM模型: {cnn_lstm_model_path}")

        self.device = torch.device('cuda' if torch.cuda.is_available() else 'cpu')
        self.cnn_lstm.to(self.device)

    def predict(self, X, weights=[0.4, 0.6]):
        """
        集成预测

        Args:
            X: 输入特征
            weights: [RF权重, CNN-LSTM权重]

        Returns:
            predictions: 预测结果
            probabilities: 预测概率
        """
        # Random Forest预测
        rf_proba = self.rf_model.predict_proba(X)

        # CNN-LSTM预测
        X_tensor = torch.FloatTensor(X).to(self.device)
        with torch.no_grad():
            cnn_lstm_output = self.cnn_lstm(X_tensor)
            cnn_lstm_proba = torch.softmax(cnn_lstm_output, dim=1).cpu().numpy()

        # 加权集成
        ensemble_proba = weights[0] * rf_proba + weights[1] * cnn_lstm_proba
        predictions = np.argmax(ensemble_proba, axis=1)

        return predictions, ensemble_proba

if __name__ == '__main__':
    # 测试集成模型
```
