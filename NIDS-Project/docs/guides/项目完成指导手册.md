```python
# traffic_capture.py
from scapy.all import sniff, IP, TCP, UDP
import threading
import queue

class TrafficCapture:
    def __init__(self, interface="eth0"):
        self.interface = interface
        self.packet_queue = queue.Queue(maxsize=10000)
        self.is_capturing = False

    def packet_callback(self, packet):
        if IP in packet:
            features = self.extract_features(packet)
            self.packet_queue.put(features)

    def extract_features(self, packet):
        # 提取网络流量特征
        return {
            'protocol': packet[IP].proto,
            'src_port': packet[TCP].sport if TCP in packet else 0,
            'dst_port': packet[TCP].dport if TCP in packet else 0,
            'packet_size': len(packet),
            'flags': packet[TCP].flags if TCP in packet else 0,
            # ... 更多特征
        }

    def start_capture(self):
        self.is_capturing = True
        sniff(iface=self.interface, prn=self.packet_callback,
              stop_filter=lambda x: not self.is_capturing)
```

```python
# flow_features.py
class FlowFeatureExtractor:
    def __init__(self, window_size=60):  # 60秒窗口
        self.window_size = window_size
        self.flows = {}  # 存储活跃流

    def extract_flow_features(self, packets):
        features = []
        for flow_key, flow_packets in self.group_by_flow(packets):
            flow_features = {
                'duration': self.calculate_duration(flow_packets),
                'total_fwd_packets': self.count_forward_packets(flow_packets),
                'total_bwd_packets': self.count_backward_packets(flow_packets),
```

```python
# real_time_detector.py
import asyncio
from concurrent.futures import ThreadPoolExecutor

class RealTimeDetector:
    def __init__(self):
        self.models = {
            'rf': self.load_model('random_forest'),
            'cnn': self.load_model('cnn'),
            'lstm': self.load_model('lstm'),
            'hybrid': self.load_model('cnn_lstm')
        }
        self.feature_extractor = FlowFeatureExtractor()
        self.response_engine = ResponseEngine()

    async def detect_loop(self):
        while True:
            # 获取最新流量数据
            packets = await self.get_recent_packets()

            # 特征提取
            features = self.feature_extractor.extract_flow_features(packets)

            # 多模型预测
            predictions = {}
            for model_name, model in self.models.items():
                predictions[model_name] = model.predict(features)

            # 集成决策
            final_prediction = self.ensemble_predict(predictions)

            # 响应处理
            if final_prediction['is_attack']:
                await self.response_engine.handle_attack(final_prediction)

            await asyncio.sleep(1)  # 1秒检测间隔
```

```python
# response_engine.py
class ResponseEngine:
    def __init__(self):
        self.response_strategies = {
            'ddos': self.handle_ddos,
            'port_scan': self.handle_port_scan,
            'brute_force': self.handle_brute_force,
            'malware': self.handle_malware
        }

    async def handle_attack(self, prediction):
        attack_type = prediction['attack_type']
        confidence = prediction['confidence']
        source_ip = prediction['source_ip']

        if confidence > 0.8:  # 高置信度
            # 立即阻断
            await self.block_ip(source_ip)
            await self.log_incident(prediction)
            await self.notify_admin(prediction)
        elif confidence > 0.6:  # 中等置信度
            # 限制访问
            await self.rate_limit_ip(source_ip)
            await self.increase_monitoring(source_ip)
        else:  # 低置信度
            # 仅记录
            await self.log_suspicious_activity(prediction)
```

```python
class ExperimentRunner:
    def __init__(self):
        self.models = ['rf', 'cnn', 'lstm', 'hybrid']
        self.datasets = ['nsl-kdd', 'cicids2017', 'unsw-nb15']
        self.metrics = ['accuracy', 'precision', 'recall', 'f1', 'auc']

    def run_baseline_comparison(self):
        results = {}
        for dataset in self.datasets:
            results[dataset] = {}
            for model in self.models:
                results[dataset][model] = self.evaluate_model(model, dataset)
        return results

    def run_ablation_study(self):
        # 消融实验：逐步移除模型组件，观察性能变化

        results = {}
```
